## 10.12 调试技巧

### 打印结构体

```rust
#[derive(Debug)]
struct Person {
    name: String,
    age: u32,
}

fn main() {
    let p = Person {
        name: String::from("Alice"),
        age: 25,
    };

    // 基本调试输出
    println!("{:?}", p);

    // 美化格式
    println!("{:#?}", p);

    // 使用 dbg!
    dbg!(&p);
}
```

### 自定义 Debug 输出

```rust
struct Password(String);

impl std::fmt::Debug for Password {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        // 不显示实际密码
        write!(f, "Password(****)")
    }
}

fn main() {
    let pwd = Password(String::from("secret123"));
    println!("{:?}", pwd);  // Password(****)
}
```

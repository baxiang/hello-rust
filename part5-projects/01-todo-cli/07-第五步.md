## 第五步：CLI 入口

### src/main.rs

```rust
use clap::{Parser, Subcommand};
use anyhow::Result;

mod todo;
mod storage;
mod commands;

use storage::Storage;

#[derive(Parser)]
#[command(name = "todo")]
#[command(author = "Your Name")]
#[command(version = "1.0")]
#[command(about = "待办事项管理工具", long_about = None)]
struct Cli {
    #[command(subcommand)]
    command: Commands,
}

#[derive(Subcommand)]
enum Commands {
    /// 添加待办事项
    Add {
        /// 待办事项内容
        content: String,
    },
    /// 列出待办事项
    List {
        /// 显示已完成的事项
        #[arg(short, long, default_value_t = false)]
        all: bool,
    },
    /// 标记完成
    Done {
        /// 待办事项 ID
        id: u32,
    },
    /// 删除待办事项
    Remove {
        /// 待办事项 ID
        id: u32,
    },
    /// 清空已完成的事项
    Clear,
}

fn main() -> Result<()> {
    let cli = Cli::parse();
    let storage = Storage::new();
    
    match cli.command {
        Commands::Add { content } => {
            commands::add(&storage, content)?;
        }
        Commands::List { all } => {
            commands::list(&storage, all)?;
        }
        Commands::Done { id } => {
            commands::done(&storage, id)?;
        }
        Commands::Remove { id } => {
            commands::remove(&storage, id)?;
        }
        Commands::Clear => {
            commands::clear_done(&storage)?;
        }
    }
    
    Ok(())
}
```

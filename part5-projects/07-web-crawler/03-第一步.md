## 第一步：创建项目

```bash
cargo new web-crawler
cd web-crawler
```

### Cargo.toml

```toml
[package]
name = "web-crawler"
version = "0.1.0"
edition = "2021"

[dependencies]
tokio = { version = "1", features = ["full"] }
reqwest = { version = "0.11", features = ["json"] }
scraper = "0.18"  # HTML 解析
url = "2"
select = "0.6"    # CSS 选择器
serde = { version = "1", features = ["derive"] }
serde_json = "1"
toml = "0.8"
anyhow = "1"
tracing = "0.1"
tracing-subscriber = "0.3"
chrono = { version = "0.4", features = ["serde"] }
futures = "0.3"
async-channel = "2"
```

## 25.11 常见错误

### 错误 1：优先级问题

```rust
macro_rules! double {
    ($e:expr) => {
        $e * $e  // ❌ 优先级问题
    };
}

fn main() {
    // 展开：2 + 3 * 2 + 3 = 11（不是 25）
    let x = double!(2 + 3);
}

// ✅ 修复
macro_rules! double_fixed {
    ($e:expr) => {
        ($e) * ($e)
    };
}
```

### 错误 2：变量捕获

```rust
macro_rules! bad_macro {
    ($x:expr) => {
        let x = $x;  // ❌ 可能捕获意外的 x
    };
}

// ✅ 使用唯一变量名
macro_rules! good_macro {
    ($x:expr) => {
        let __macro_x = $x;
    };
}
```

### 错误 3：片段类型错误

```rust
macro_rules! wrong_type {
    ($x:ident) => { /* ... */ };
}

fn main() {
    // ❌ 错误：123 不是 ident
    // wrong_type!(123);

    // ✅ 正确
    wrong_type!(my_var);
}
```

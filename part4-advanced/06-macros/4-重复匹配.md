## 25.4 重复匹配

### * 和 + 重复

```rust
// 匹配零次或多次（*）
macro_rules! vec_of_strings {
    ($($x:expr),*) => {
        vec![$(String::from($x)),*]
    };
}

// 匹配一次或多次（+）
macro_rules! one_or_more {
    ($($x:expr),+) => {
        vec![$($x),+]
    };
}

fn main() {
    let v = vec_of_strings!["hello", "world", "rust"];
    println!("{:?}", v);

    let v2 = one_or_more![1, 2, 3];
    println!("{:?}", v2);
}
```

### 分隔符

```rust
macro_rules! multi_match {
    // 逗号分隔
    ($($x:expr),*) => { /* 1, 2, 3 */ };

    // 分号分隔
    ($($x:expr);*) => { /* 1; 2; 3 */ };

    // 空格分隔
    ($($x:expr) *) => { /* 1 2 3 */ };

    // 无分隔符
    ($($x:tt)*) => { /* 任意 tokens */ };
}
```

### 重复嵌套

```rust
macro_rules! matrix {
    ($($( $val:expr ),*);*) => {
        {
            println!("矩阵：");
            $(
                $(
                    print!("{} ", $val);
                )*
                println!();
            )*
        }
    };
}

fn main() {
    matrix! {
        1, 2, 3;
        4, 5, 6;
        7, 8, 9
    }
}
```

## 28.2 断言宏

### 常用断言

```rust
#[test]
fn test_assertions() {
    // assert! - 条件为真
    assert!(true);
    assert!(2 + 2 == 4);

    // assert! 带错误消息
    assert!(2 + 2 == 4, "数学出错了");

    // assert_eq! - 相等
    assert_eq!(10, 5 + 5);
    assert_eq!("hello", "hel".to_string() + "lo");

    // assert_ne! - 不相等
    assert_ne!(1, 2);
    assert_ne!("hello", "world");

    // 自定义错误消息
    assert_eq!(2 + 2, 4, "加法运算失败");
}
```

### 测试 Result 和 Option

```rust
#[test]
fn test_result() -> Result<(), String> {
    if 2 + 2 == 4 {
        Ok(())
    } else {
        Err("数学出错了".to_string())
    }
}

#[test]
fn test_option() {
    let value: Option<i32> = Some(5);

    // is_some / is_none
    assert!(value.is_some());
    assert!(!value.is_none());

    // unwrap
    assert_eq!(value.unwrap(), 5);
}
```

### 测试 panic

```rust
#[test]
#[should_panic]
fn test_panic() {
    panic!("预期的 panic");
}

#[test]
#[should_panic(expected = "预期的 panic")]
fn test_panic_message() {
    panic!("预期的 panic");
}

#[test]
fn test_panic_result() {
    let result = std::panic::catch_unwind(|| {
        panic!("test panic");
    });
    assert!(result.is_err());
}
```

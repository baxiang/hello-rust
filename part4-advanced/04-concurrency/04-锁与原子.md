## 23.7 原子类型

### 无锁并发

```rust
use std::sync::atomic::{AtomicUsize, Ordering};
use std::sync::Arc;
use std::thread;

fn main() {
    let counter = Arc::new(AtomicUsize::new(0));
    let mut handles = vec![];

    for _ in 0..10 {
        let counter = Arc::clone(&counter);
        let handle = thread::spawn(move || {
            counter.fetch_add(1, Ordering::SeqCst);
        });
        handles.push(handle);
    }

    for h in handles {
        h.join().unwrap();
    }

    println!("结果：{}", counter.load(Ordering::SeqCst));
}
```

### 常用原子操作

```rust
use std::sync::atomic::{AtomicBool, AtomicUsize, Ordering};

fn main() {
    // 原子整数
    let atomic = AtomicUsize::new(0);

    // 加载
    let val = atomic.load(Ordering::SeqCst);

    // 存储
    atomic.store(10, Ordering::SeqCst);

    // 交换
    let old = atomic.swap(20, Ordering::SeqCst);

    // 比较并交换
    atomic.compare_exchange(20, 30, Ordering::SeqCst, Ordering::SeqCst);

    // 获取并加
    atomic.fetch_add(5, Ordering::SeqCst);
    atomic.fetch_sub(3, Ordering::SeqCst);
    atomic.fetch_and(0xF, Ordering::SeqCst);
    atomic.fetch_or(0x10, Ordering::SeqCst);

    // 原子布尔
    let flag = AtomicBool::new(false);
    flag.store(true, Ordering::SeqCst);
    let val = flag.load(Ordering::SeqCst);
}
```

### 内存序

```rust
use std::sync::atomic::Ordering;

// 内存序选项：
// Relaxed     - 只保证原子性，无顺序保证
// Release     - 释放操作，前面的操作不能被重排到后面
// Acquire     - 获取操作，后面的操作不能被重排到前面
// AcqRel      - 获取 + 释放
// SeqCst      - 顺序一致（默认，最严格）
```

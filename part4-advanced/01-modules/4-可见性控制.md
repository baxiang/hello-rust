## 20.4 可见性控制

### pub 关键字

```rust
// 默认私有：模块外不可见
mod private_mod {
    fn private_func() {}          // 私有函数
    pub fn public_func() {}       // 模块内可见
}

// 公共模块
pub mod public_mod {
    pub fn public_func() {}       // 外部可见
    fn private_func() {}          // 模块内可见
}

// 结构体可见性
struct PrivateStruct {            // 私有结构体
    pub x: i32,                   // 但结构体私有，字段也无效
}

pub struct PublicStruct {         // 公共结构体
    pub x: i32,                   // 外部可访问
    y: i32,                       // 包内可访问（默认）
    pub(crate) z: i32,            // 整个 crate 内可见
}
```

### 可见性修饰符

```rust
// pub(crate)：整个 crate 内可见
pub(crate) struct CrateWide {
    // 只能在当前 crate 内访问
}

// pub(super)：父模块内可见
pub(super) struct ParentOnly {
    // 只能在父模块访问
}

// pub(in path)：指定路径内可见
pub(in crate::module) struct ModuleOnly {
    // 只能在 crate::module 内访问
}

// 示例
mod outer {
    pub(super) fn visible_to_parent() {}

    mod inner {
        // 可以访问
        super::visible_to_parent();
    }
}
```

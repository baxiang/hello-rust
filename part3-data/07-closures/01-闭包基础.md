## 18.2 闭包语法基础

### 基本语法

```rust
fn main() {
    // 最简单的闭包：参数 | 表达式
    let add_one = |x| x + 1;
    let result = add_one(5);
    println!("add_one(5) = {}", result);  // 6

    // 闭包的完整形式
    let add = |x: i32, y: i32| -> i32 {
        x + y
    };
    println!("add(3, 5) = {}", add(3, 5));  // 8
}
```

### 多种写法对比

```rust
fn main() {
    // 1. 无参数
    let say_hello = || println!("Hello!");
    say_hello();

    // 2. 单参数（最常见）
    let double = |x| x * 2;
    println!("{}", double(5));  // 10

    // 3. 多参数
    let add = |x, y| x + y;
    println!("{}", add(3, 5));  // 8

    // 4. 多行闭包（使用花括号）
    let add_with_logging = |x, y| {
        println!("计算：{} + {}", x, y);
        let result = x + y;
        println!("结果：{}", result);
        result  // 最后一行是返回值
    };
    println!("{}", add_with_logging(10, 20));

    // 5. 显式类型标注
    let typed_add: fn(i32, i32) -> i32 = |x: i32, y: i32| -> i32 {
        x + y
    };

    // 6. 只标注参数类型
    let param_typed = |x: i32, y: i32| x + y;

    // 7. 只标注返回类型
    let return_typed = |x, y| -> i32 { x + y };
}
```

### 闭包 vs 函数

```rust
// ========== 函数 ==========
fn add_one_fn(x: i32) -> i32 {
    x + 1
}

// ========== 闭包 ==========
let add_one_closure = |x: i32| -> i32 { x + 1 };

// ========== 对比 ==========
// 函数：
// • 有名字
// • 必须标注参数和返回类型
// • 可以递归调用
// • 在作用域内任意位置可见

// 闭包：
// • 通常匿名（可以绑定到变量）
// • 可以类型推断
// • 不能直接递归（需要特殊处理）
// • 可以捕获环境变量

fn main() {
    // 使用函数
    println!("{}", add_one_fn(5));

    // 使用闭包
    println!("{}", add_one_closure(5));

    // 函数名可以作为函数指针传递
    let func_ptr: fn(i32) -> i32 = add_one_fn;
    println!("{}", func_ptr(10));

    // 闭包类型是唯一的、不可命名的
    // 必须用泛型或 trait 对象传递
}
```

## 15.8 完整示例

### 示例 1：通用栈数据结构

```rust
use std::fmt::Display;

/// 泛型栈结构
#[derive(Debug)]
struct Stack<T> {
    items: Vec<T>,
}

impl<T> Stack<T> {
    /// 创建空栈
    fn new() -> Self {
        Stack { items: Vec::new() }
    }

    /// 压栈
    fn push(&mut self, item: T) {
        self.items.push(item);
    }

    /// 弹栈
    fn pop(&mut self) -> Option<T> {
        self.items.pop()
    }

    /// 查看栈顶
    fn peek(&self) -> Option<&T> {
        self.items.last()
    }

    /// 栈是否为空
    fn is_empty(&self) -> bool {
        self.items.is_empty()
    }

    /// 栈大小
    fn len(&self) -> usize {
        self.items.len()
    }

    /// 清空栈
    fn clear(&mut self) {
        self.items.clear();
    }
}

// 只有 T 实现 Clone 时才能克隆栈
impl<T: Clone> Stack<T> {
    fn clone(&self) -> Self {
        Stack {
            items: self.items.clone(),
        }
    }
}

// 只有 T 实现 Display 时才能打印栈内容
impl<T: Display> Stack<T> {
    fn print(&self) {
        println!("[");
        for item in &self.items {
            println!("  {}", item);
        }
        println!("]");
    }
}

// 默认实现
impl<T> Default for Stack<T> {
    fn default() -> Self {
        Self::new()
    }
}

fn main() {
    // 整数栈
    let mut int_stack: Stack<i32> = Stack::new();
    int_stack.push(1);
    int_stack.push(2);
    int_stack.push(3);

    println!("整数栈:");
    int_stack.print();
    println!("栈顶：{:?}", int_stack.peek());
    println!("弹出：{:?}", int_stack.pop());

    // 字符串栈
    let mut string_stack: Stack<String> = Stack::new();
    string_stack.push(String::from("Hello"));
    string_stack.push(String::from("World"));

    println!("\n字符串栈:");
    string_stack.print();

    while let Some(item) = string_stack.pop() {
        println!("弹出：{}", item);
    }
}
```

### 示例 2：泛型容器与迭代器

```rust
use std::fmt::Display;

/// 泛型容器
#[derive(Debug)]
struct Container<T> {
    items: Vec<T>,
}

impl<T> Container<T> {
    fn new() -> Self {
        Container { items: Vec::new() }
    }

    fn add(&mut self, item: T) {
        self.items.push(item);
    }

    fn get(&self, index: usize) -> Option<&T> {
        self.items.get(index)
    }

    fn len(&self) -> usize {
        self.items.len()
    }

    fn is_empty(&self) -> bool {
        self.items.is_empty()
    }

    fn iter(&self) -> impl Iterator<Item = &T> {
        self.items.iter()
    }
}

// 条件实现：只有 T 实现 PartialEq 时
impl<T: PartialEq> Container<T> {
    fn contains(&self, item: &T) -> bool {
        self.items.contains(item)
    }

    fn remove(&mut self, item: &T) -> bool {
        if let Some(pos) = self.items.iter().position(|x| x == item) {
            self.items.remove(pos);
            true
        } else {
            false
        }
    }
}

// 条件实现：只有 T 实现 Display 时
impl<T: Display> Container<T> {
    fn print_all(&self) {
        for (i, item) in self.items.iter().enumerate() {
            println!("{}: {}", i, item);
        }
    }
}

// 条件实现：只有 T 实现 Ord 时
impl<T: Ord> Container<T> {
    fn sort(&mut self) {
        self.items.sort();
    }

    fn min(&self) -> Option<&T> {
        self.items.iter().min()
    }

    fn max(&self) -> Option<&T> {
        self.items.iter().max()
    }
}

// 条件实现：只有 T 实现 Clone 时
impl<T: Clone> Container<T> {
    fn to_vec(&self) -> Vec<T> {
        self.items.clone()
    }
}

fn main() {
    let mut container = Container::new();
    container.add(5);
    container.add(2);
    container.add(8);
    container.add(1);
    container.add(9);

    println!("原始数据:");
    container.print_all();

    // 排序
    container.sort();
    println!("\n排序后:");
    container.print_all();

    println!("\n最小值：{:?}", container.min());
    println!("最大值：{:?}", container.max());
    println!("包含 5: {}", container.contains(&5));

    container.remove(&5);
    println!("\n移除 5 后:");
    container.print_all();
}
```

### 示例 3：泛型算法

```rust
// 二分查找
fn binary_search<T: Ord>(slice: &[T], target: &T) -> Option<usize> {
    let mut left = 0;
    let mut right = slice.len();

    while left < right {
        let mid = left + (right - left) / 2;
        match slice[mid].cmp(target) {
            std::cmp::Ordering::Equal => return Some(mid),
            std::cmp::Ordering::Less => left = mid + 1,
            std::cmp::Ordering::Greater => right = mid,
        }
    }

    None
}

// 快速排序分区
fn partition<T: Ord>(arr: &mut [T], low: usize, high: usize) -> usize {
    let pivot = high;
    let mut i = low;

    for j in low..high {
        if arr[j] <= arr[pivot] {
            arr.swap(i, j);
            i += 1;
        }
    }

    arr.swap(i, high);
    i
}

// 泛型冒泡排序
fn bubble_sort<T: Ord>(arr: &mut [T]) {
    let len = arr.len();
    for i in 0..len {
        for j in 0..len - i - 1 {
            if arr[j] > arr[j + 1] {
                arr.swap(j, j + 1);
            }
        }
    }
}

fn main() {
    // 二分查找
    let sorted = [1, 3, 5, 7, 9, 11];
    println!("查找 7: {:?}", binary_search(&sorted, &7));
    println!("查找 4: {:?}", binary_search(&sorted, &4));

    // 冒泡排序
    let mut numbers = [64, 34, 25, 12, 22, 11, 90];
    bubble_sort(&mut numbers);
    println!("排序后：{:?}", numbers);

    // 字符串排序
    let mut strings = ["banana", "apple", "cherry", "date"];
    bubble_sort(&mut strings);
    println!("字符串排序：{:?}", strings);
}
```

## 15.9 常见错误

### 错误 1：缺少必要的约束

```rust
// ❌ 错误：T 没有实现 Display
// fn print_item<T>(item: T) {
//     println!("{}", item);  // 编译错误！
// }

// ✅ 正确
fn print_item<T: std::fmt::Display>(item: T) {
    println!("{}", item);
}
```

### 错误 2：类型参数不一致

```rust
struct Point<T> {
    x: T,
    y: T,
}

fn main() {
    // ❌ 错误：x 和 y 必须是同一类型
    // let p = Point { x: 5, y: 4.0 };

    // ✅ 正确：使用两个泛型参数
    let p = Point { x: 5, y: 4.0 };
}

// 修正版本
struct Point<T, U> {
    x: T,
    y: U,
}
```

### 错误 3：在 struct 定义时添加约束

```rust
// ❌ 错误：不要在结构体定义中添加不必要的约束
// struct Point<T: Copy> {
//     x: T,
//     y: T,
// }

// ✅ 正确：约束应该在 impl 块中
struct Point<T> {
    x: T,
    y: T,
}

impl<T: Copy> Point<T> {
    fn get_x(&self) -> T {
        self.x  // Copy 允许直接复制
    }
}
```

### 错误 4：泛型与生命周期

```rust
// ❌ 错误：忽略生命周期
// fn get_first<T>(slice: &[T]) -> &T {
//     &slice[0]
// }

// ✅ 正确：显式声明生命周期
fn get_first<'a, T>(slice: &'a [T]) -> &'a T {
    &slice[0]
}

// 或者让编译器推断
fn get_first_auto<T>(slice: &[T]) -> &T {
    &slice[0]
}
```

### 错误 5：滥用泛型

```rust
// ❌ 不推荐：过度泛化，代码复杂
fn process<T, U, V, W>(t: T, u: U, v: V, w: W) -> String
where
    T: Clone + Display,
    U: Into<String>,
    V: From<T>,
    W: PartialEq<V>,
{
    String::new()
}

// ✅ 推荐：简单明确
fn process(input: &str) -> String {
    format!("Processed: {}", input)
}
```

## 15.11 小结

### 泛型核心概念

| 概念 | 说明 |
|------|------|
| 泛型函数 | `fn foo<T>(x: T) -> T` |
| 泛型结构体 | `struct Foo<T> { x: T }` |
| 泛型枚举 | `enum Foo<T> { Bar(T) }` |
| Trait 约束 | `T: Trait1 + Trait2` |
| where 子句 | 复杂的约束条件 |
| 单态化 | 编译时生成专用代码 |

### 常用约束

| Trait | 用途 |
|-------|------|
| `Display` | 格式化输出 |
| `Debug` | 调试输出 |
| `Clone` | 克隆 |
| `Copy` | 复制 |
| `PartialOrd` | 部分排序 |
| `PartialEq` | 相等比较 |
| `Eq` | 完全相等 |
| `Ord` | 完全排序 |

### 最佳实践

1. **在函数/方法上添加约束**，而不是结构体定义
2. **使用 where 子句**处理复杂约束
3. **保持简单**：不要过度泛化
4. **考虑性能**：单态化可能导致代码膨胀
5. **提供有用的错误信息**：使用 expect 而不是 unwrap
